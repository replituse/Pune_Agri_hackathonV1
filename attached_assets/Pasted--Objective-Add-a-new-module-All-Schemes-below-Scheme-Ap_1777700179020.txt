🔥 FULL CODE (React + Tailwind)
import React, { useEffect, useState } from "react";

const PAGE_SIZE = 6;

const AllSchemes = () => {
  const [schemes, setSchemes] = useState([]);
  const [view, setView] = useState("table");
  const [page, setPage] = useState(1);

  useEffect(() => {
    fetchSchemes();
  }, []);

  const fetchSchemes = async () => {
    // Replace with API
    const data = await fetch("/api/schemes").then(res => res.json());
    setSchemes(data);
  };

  const paginated = schemes.slice(
    (page - 1) * PAGE_SIZE,
    page * PAGE_SIZE
  );

  return (
    <div className="p-6">
      {/* Header */}
      <div className="flex justify-between mb-4">
        <h1 className="text-2xl font-bold">All Schemes</h1>

        <div className="flex gap-2">
          <button onClick={() => setView("table")} className="btn">
            Table
          </button>
          <button onClick={() => setView("grid")} className="btn">
            Grid
          </button>
        </div>
      </div>

      {/* TABLE VIEW */}
      {view === "table" && (
        <table className="w-full border rounded-lg">
          <thead className="bg-gray-100">
            <tr>
              <th>Name</th>
              <th>Type</th>
              <th>Category</th>
              <th>Eligibility</th>
            </tr>
          </thead>
          <tbody>
            {paginated.map((s, i) => (
              <tr key={i} className="border-t">
                <td>{s.name}</td>
                <td>{s.type}</td>
                <td>{s.category}</td>
                <td>{s.eligibility?.summary}</td>
              </tr>
            ))}
          </tbody>
        </table>
      )}

      {/* GRID VIEW */}
      {view === "grid" && (
        <div className="grid grid-cols-3 gap-4">
          {paginated.map((s, i) => (
            <div key={i} className="p-4 border rounded-lg shadow">
              <h2 className="font-bold">{s.name}</h2>
              <p>{s.type}</p>
              <p className="text-sm">{s.description}</p>
            </div>
          ))}
        </div>
      )}

      {/* PAGINATION */}
      <div className="flex justify-center mt-6 gap-2">
        {Array.from({ length: Math.ceil(schemes.length / PAGE_SIZE) })
          .map((_, i) => (
            <button
              key={i}
              onClick={() => setPage(i + 1)}
              className={`px-3 py-1 border ${
                page === i + 1 ? "bg-green-600 text-white" : ""
              }`}
            >
              {i + 1}
            </button>
          ))}
      </div>
    </div>
  );
};

export default AllSchemes;
